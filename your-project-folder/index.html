function displayFilteredWords(words) { filteredWordsContainer.innerHTML = `
${words.map(word => `` ).join('')}
`; document.querySelectorAll('.word').forEach(button => { button.addEventListener('click', () => { const word = button.textContent; selectedWordDisplay.textContent = word; filteredWordsContainer.style.display = 'none'; selectedWordDisplay.style.display = 'block'; // Small delay to ensure the word is displayed before redirecting setTimeout(() => { // Create the Google Image search URL with lowercase word const searchUrl = `https://www.google.com/search?q=${encodeURIComponent(word.toLowerCase())}&tbm=isch`; // Replace current history entry with a blank one history.replaceState(null, '', 'about:blank'); // Navigate to Google Image search in the same tab window.location.href = searchUrl; }, 100); }); }); filteredWordsContainer.style.display = 'block'; }
